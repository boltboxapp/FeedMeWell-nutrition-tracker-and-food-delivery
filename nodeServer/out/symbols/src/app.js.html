230 *  Home page is where the user sees
231 *  their stats.
232 *  In order to be able to see the stats,

240 /**
241  * Checks if the user is logged in, if not the user is then redirected to where necessary
242  * @param {object} req - The requests for the needed credentials
243  * @param {object} res - Redirects to where necessary depending on the lack of credentials.
244  */
245 app.get('/', isLoggedIn, function(req, res){
246   if (!req.user.address){
247     res.redirect("/address");
248   }
249   else if (!req.user.tokens.fitbit.access) { // user is not loged into fitbit account (no refresh token)
250     res.redirect(OAuth2Urls.fitbitAuthUrl);
251   }
252   else if (!req.user.tokens.delivery.access) { // user is not loged into delivery account
253     res.redirect(OAuth2Urls.deliveryAuthUrl);
254   }
255   routes.index(req, res);
256 });

296 /**
297  * Fitbit authentication for the fitbit.
298  * @param {object} req - The requests for the needed credentials
299  * @param {object} res - Redirects to where necessary depending on the lack of credentials.
300  */
301 app.get('/auth/fitbit', function(req, res){
302   if (req.query.code){ // about to request tokens
303     try{
304         console.log(req.query.code); // get the code from the callback url! (how convenient!)
305       res.header({
306         "Authorization": OAuth2Urls.fitbitHeaders["Authorization"], 
307         "Content-Type": OAuth2Urls.fitbitHeaders["Content-Type"] 
308       });
309 
310       // POSTing from the Server side
311 	  /**
312 	   * Posts from the server side requesting the needed information
313 	   * @var {object} options
314 	   */
315       var options = {
316         headers: {
317           "Authorization": OAuth2Urls.fitbitHeaders["Authorization"], 
318           "Content-Type": OAuth2Urls.fitbitHeaders["Content-Type"] 
319         }
320       };
321       needle
322         .post(OAuth2Urls.fitbitTokenUrl + req.query.code, {}, options, function(err, resp){
323           if (err) {
324             res.send(err);
325           }
326           console.log("[AFTER REQUESTING TOKENS] BODY: " + JSON.stringify(resp.body));
327           User.update({"google.id":req.user.google.id}, {$set: {"tokens.fitbit.access": resp.body.access_token}}, function(err){
328             res.send(err);
329           });
330           User.update({"google.id":req.user.google.id}, {$set: {"tokens.fitbit.refresh": resp.body.refresh_token}}, function(err){
331             res.send(err);
332           });
333           
334           res.redirect('/');
335         }).on('end', function(err, resp){
336           if(err){
337             res.send(err);
338           }
339         });
340     }
341     catch(err){
342       res.send(err);
343     }
344   }
345   else { // tokens already requested
346     res.send(req.body);
347   }
348 });
