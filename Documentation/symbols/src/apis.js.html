 72 /**
 73  * Searches recursively for the right child's data
 74  * @param {number} iterable Number of iterations
 75  * @param {object} callback
 76  */
 77 function recursiveSearch(iterable, callback){
 78     if (iterable.children.length !== 0){
 79         //console.log(iterable);
 80         $.each(iterable.children, function(value){
 81            recursiveSearch(iterable.children[value], callback); 
 82         });
 83     }
 84     else {
 85         callback(iterable);
 86     }
 87 }
