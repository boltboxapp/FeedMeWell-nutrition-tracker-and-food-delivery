1 /**
  2  * Function that extracts paramaters from url, ex: 10 will extract 10 and so on.
  3  * e.g. url.html?parameter=10, it will extract 10 from parameter
  4  * @param {integer} sParam The number of parameters to extract.
  5  */
  6 function GetURLParameter(sParam) // function that extracts parameters from url (e.g. url.html?parameter=10, it will extract 10 from parameter)
  7 	{
  8 		/** The default page url. */
  9 	    var sPageURL = window.location.search.substring(1);
 10 	    /** The default page url parsed in. */
 11 		var sURLVariables = sPageURL.split('&');
 12 	    for (var i = 0; i < sURLVariables.length; i++)
 13 	    {
 14 			/** Checks for a matching url hit. */
 15 	        var sParameterName = sURLVariables[i].split('=');
 16 	        if (sParameterName[0] == sParam)
 17 	        {
 18 	        	console.log(sParameterName);
 19 	            return sParameterName[1];
 20 	        }
 21 	    }
 22 	}
